	/**Initialization for Table B**/
	private void initializeTableB(){
		String[] countries1=DataReader.getCountries(true);
		Helper.initDatePickers(null, datePickerTableB);
		Helper.resetSelection(DeathCountrySelection,countries1);
		var h = CasesAnalysis.getContinents(countries1);
		deathTableContinent.getItems().addAll("All", "Asia","Africa","Europe","North America",
				"South America","Oceania");
		deathTableContinent.getSelectionModel().select("All");
		ArrayList<String> countryName = new ArrayList<String>();
		deathTableContinent.setOnAction((e)->{
			countryName.clear();
    		var getContinent = (String)deathTableContinent.getSelectionModel().getSelectedItem();
    		
    		if (getContinent.compareTo("All")==0) {
    			for (var c:countries1) {
    				countryName.add(c);
    			}
    		} else {
    			for (var c:countries1) {
    				if (h.get(c).equals(getContinent)) {
    					countryName.add(c);
    				}
    			}
    		}
    		String[] countries;
    		countries = new String[countryName.size()];
			countryName.toArray(countries);
			Helper.resetSelection(DeathCountrySelection,countries);
    	});
	

    	datePickerTableB.setValue(LocalDate.of(2021, 7, 20));
    	datePickerTableB.getEditor().setDisable(true);
    	datePickerTableB.getEditor().setOpacity(1);

    	checkboxDeath.setSelected(false);
    	checkboxDeath.setOnAction((e)->{
    		if (checkboxDeath.isSelected()) {
    			datePickerTableB.setValue(null);
    			datePickerTableB.getEditor().setDisable(false);
    		} else {
    			datePickerTableB.setValue(LocalDate.of(2021, 7, 20));
    	    	datePickerTableB.getEditor().setDisable(true);
    	    	datePickerTableB.getEditor().setOpacity(1);
    		}
    	});
	}

    		chart=(LineChart)arr[0];

    	}

    		e.printStackTrace();
    	}
    	consoleLog(error);
    	final String errorMsg=error;
    	final LineChart chartViewed=chart;
    	Platform.runLater(()->{
    		try {
    			if(errorMsg.isEmpty()==false)
    				Thread.sleep(1000);
			} catch (InterruptedException e) {
				e.printStackTrace();
			}
    		DataViewer.viewChart(chartViewed);
    	});

    		date1 = ChartDeathDate1.getEditor().getText();

    		LocalDate localDate2 = ChartDeathDate2.getValue();

        	DateFormat dateFormat2 = new SimpleDateFormat("MM/dd/yyyy");  
        	date2 = dateFormat2.format(dateB);
    	} catch(Exception e) {
    		date2 = ChartDeathDate2.getEditor().getText();
    	}
    	
    	Date checking1, checking2;
    	// Validate Date1
	   	try {
	   		SimpleDateFormat sdf1 = new SimpleDateFormat("MM/dd/yyyy");
	   		sdf1.setLenient(false);
	        checking1 = sdf1.parse(date1);
	   	} catch (Exception e) {
	       	 System.out.print(date1);
	       	 Alert alert1 = new Alert(AlertType.ERROR, "Invalid Date Format!");
	       	 alert1.setContentText("Please Enter (Record From) Again in MM/dd/yyyy format!");
	       	 alert1.show();
	       	 return;
	     }
	   	
	   	// Validate Date2
	   	try {
	   		SimpleDateFormat sdf2 = new SimpleDateFormat("MM/dd/yyyy");
	   		sdf2.setLenient(false);
	        checking2 = sdf2.parse(date2);
	   	} catch (Exception e) {
	       	 Alert alert2 = new Alert(AlertType.ERROR, "Invalid Date Format!");
	       	 alert2.setContentText("Please Enter (Record To) Again in MM/dd/yyyy format!");
	       	 alert2.show();
	       	 return;
	     }
	   	
	   	// Check if "date from" is more than date to
	   	if(checking2.before(checking1)) {
	   		Alert alert3 = new Alert(AlertType.ERROR, "Impossible Record Format!");
	       	alert3.setContentText("Please Ensure that Record is Valid!");
	       	alert3.show();
    		return;
    	}
	   	
	   	
    	LineChart chart = Chart.getDeathChart1(countryArray, date1, date2);
    	DataViewer.viewChart(chart);
    }
