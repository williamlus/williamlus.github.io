package comp3111.covid;

import javafx.geometry.Pos;
import javafx.scene.Node;
import javafx.scene.Scene;
import javafx.scene.layout.HBox;
import javafx.scene.layout.VBox;
import javafx.stage.Stage;

/** A class to visualize graphs.
 * */
public class DataViewer {
	
	public static void viewTable(Node table) {
		Stage stage = new Stage();
        VBox vbox = new VBox();
        vbox.setAlignment(Pos.CENTER);
        vbox.getChildren().addAll(table);
        Scene scene = new Scene(vbox);
        stage.setScene(scene);
        stage.show();
	}
	
	public static void viewChart(Node chart) {
		Stage stage = new Stage();
        VBox vbox = new VBox();
        vbox.setAlignment(Pos.CENTER);
        vbox.getChildren().addAll(chart);
        Scene scene = new Scene(vbox);
        stage.setScene(scene);
        stage.show();
	}
	
	public static void viewPieChart(Node chart1,Node chart2) {
		Stage stage = new Stage();
		HBox hbox=new HBox(chart1, chart2);
		hbox.setSpacing(10);
        Scene scene = new Scene(hbox);
        stage.setScene(scene);
        stage.show();
	}
	
	public static void viewBarChart(Node chart1,Node chart2) {
		Stage stage = new Stage();
        VBox vbox = new VBox();
        vbox.setAlignment(Pos.CENTER);
        vbox.getChildren().addAll(chart1,chart2);
        vbox.autosize();
        Scene scene = new Scene(vbox);
        stage.setScene(scene);
        stage.show();
	}
}
