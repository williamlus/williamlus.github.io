	/**
	 * Returns LineChart of Confirmed cases and additional message.
	 * @param countries the list of countries
	 * @param startDate the start date of period of interest
	 * @param endDate the end date of period of interest
	 * @return array of LineChart and String message
	 * **/
	public static Object[] getConfirmedChart(String[] countries, Date startDate, Date endDate) throws Exception{
		
		SimpleDateFormat sdf = new SimpleDateFormat("MM/dd/yyyy");
		SimpleDateFormat sdfChart = new SimpleDateFormat("MMM dd, yyyy");
		String msg="";
		
		// Set the x-axis
		final NumberAxis xAxis = new NumberAxis();
        xAxis.setLabel("Dates of Interest");
        xAxis.setTickLabelFormatter(new NumberAxis.DefaultFormatter(xAxis) {
        	@Override             
        	public String toString(final Number object){                 
        		double time=(Double)object;
        		Date date=new Date((long)time);
        		return sdfChart.format(date);
        	}         
        });
   
        // Set the y-axis
        final NumberAxis yAxis = new NumberAxis();
        yAxis.setLabel("Cumulative Confirmed Cases (per 1M)");
        
        // Create Line Chart
        LineChart<Number, Number> lineChart = new LineChart<Number, Number>(xAxis, yAxis);
        lineChart.setTitle("Cumulative Confirmed COVID-19 Cases (per 1M)");
        
        lineChart.setPrefHeight(600.0);
        lineChart.setPrefWidth(1000.0);
                
        lineChart.setCreateSymbols(false);
        xAxis.setAutoRanging(true);
        xAxis.setForceZeroInRange(false);
        yAxis.setAutoRanging(true);
        yAxis.setForceZeroInRange(false);
        lineChart.autosize();
        
        DataReader.sortByLocationDate();
        Record[] records=DataReader.getRecords();
        
        for(String country:countries) {
        	XYChart.Series list = new XYChart.Series(); // Use a new line
    		list.setName(country);
    		float totalPerM=0; // Record the latest number
  
    		int[] range=DataReader.findByLocation(country); // Find the index range
    		Date firstDate=null;
    		Date lastDate=null;
    		
    		// Records not found
    		if(range[0]==-1) {
    			list.getData().add(new XYChart.Data<>(startDate.getTime(),0));
        		list.getData().add(new XYChart.Data<>(endDate.getTime(),0));
        		lineChart.getData().add(list);
        		msg+="Warning: the plot for "+country+" is constant zero since its records are not found!\n";
        		continue;
    		}
    		else {
    			firstDate=sdf.parse(DataReader.getRecord(range[0]).get("date"));
    			lastDate=sdf.parse(DataReader.getRecord(range[1]).get("date"));
    		}
    		
    		// Find the largest number of records
    		String temp=records[range[1]].get("total_cases_per_million");
    		float totalPerMEnd=temp.length()==0?0:Float.parseFloat(temp);
    		
    		// Find the index range of period for chart
        	int[] index=DataReader.findByPeriodForChart(country, startDate, endDate);
        	int left=index[0];
        	int right=index[1];
        	
        	if(left==-1) { // startDate > largest date of records
        		list.getData().add(new XYChart.Data<>(startDate.getTime(),totalPerMEnd));
        		list.getData().add(new XYChart.Data<>(endDate.getTime(),totalPerMEnd));
        		lineChart.getData().add(list);
        		msg+="Warning: the plot for "+country+" is a constant extension of its last record on "+sdfChart.format(lastDate)+".\n";
        		continue;
        	}
        	if(right==-1) { // endDate < smallest date of records
        		list.getData().add(new XYChart.Data<>(startDate.getTime(),0));
        		list.getData().add(new XYChart.Data<>(endDate.getTime(),0));
        		lineChart.getData().add(list);
        		msg+="Warning: the plot for "+country+" is a zero extension of its first record on "+sdfChart.format(firstDate)+".\n";
        		continue;
        	}
        	
        	// Extend the chart if startDate/endDate out of range
        	if(startDate.compareTo(firstDate)<0) {
        		list.getData().add(new XYChart.Data<>(startDate.getTime(),0));
        		msg+="Warning: the plot for "+country+" has a zero extension before its first record on "+sdfChart.format(firstDate)+".\n";
        	}
        	if(endDate.compareTo(lastDate)>0) {
        		list.getData().add(new XYChart.Data<>(endDate.getTime(),totalPerMEnd));
        		msg+="Warning: the plot for "+country+" has a constant extension after its last record on "+sdfChart.format(lastDate)+".\n";
        	}
        	
        	// Plot the chart
        	for(int i=left;i<=right;i++) {
        		Record rec=records[i];
        		String s = rec.get("total_cases_per_million");
        		String dateStr=rec.get("date");
				Date date=sdf.parse(dateStr);
				if (!s.equals("")) {
					float temp2=Float.parseFloat(s);
					if(totalPerM<temp2)
						totalPerM=temp2;
					list.getData().add(new XYChart.Data<>(date.getTime(),totalPerM));
				} 
				else {
					list.getData().add(new XYChart.Data<>(date.getTime(),totalPerM));
				}
        	}
        	lineChart.getData().add(list);
        }

		return new Object[] {lineChart,msg};
	}

        lineChart.autosize();
        
        DataReader.sortByLocationDate();
        Record[] records=DataReader.getRecords();
        
        for(String country:countries) {
        	XYChart.Series list = new XYChart.Series(); // Use a new line
    		list.setName(country);
    		float totalPerM=0; // Record the latest number

    			String temp=records[po].get("people_fully_vaccinated_per_hundred");

        		lineChart.getData().add(list);
        		continue;

        	if(left==range[0]) {
        		list.getData().add(new XYChart.Data<>(startDate.getTime()-1,0));
        	}
        	if(right==range[1]) {
        		list.getData().add(new XYChart.Data<>(endDate.getTime()+1,totalPerMEnd));
        	}
        	
    		// Find the largest number before start
    		for(int po=left;po>=range[0];po--) {

    			totalPerM=temp.length()==0?0:Float.parseFloat(temp);

	/**
	 * Returns an array of the linear regression result.
	 * @param y the original data
	 * @param num the number of data in the input array
	 * **/
	public static float[] doRegression(float[] y, int num) throws Exception{
		float[] result=new float[num];
		float xbar=0f;
		float ybar=0f;
		for (int i=0;i<num;i++) {
			xbar=xbar+i;
			ybar=ybar+y[i];
		}
		xbar=xbar/num;
		ybar=ybar/num;
		float bbar=0f;
		float sq=0f;
		for (int i=0;i<num;i++) {
			float xi=i-xbar;
			float yi=y[i]-ybar;
			bbar=bbar+xi*yi;
			sq=sq+xi*xi;
		}
		bbar=bbar/sq;
		float abar=ybar-xbar*bbar;
		for (int i=0;i<num;i++) {
			result[i]=abar+bbar*i;
		}
		return result;
	}
