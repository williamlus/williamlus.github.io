package comp3111.covid;

import java.text.SimpleDateFormat;
import java.util.Date;
import java.util.HashMap;
import javafx.collections.FXCollections;
import javafx.collections.ObservableList;
import javafx.scene.Node;
import javafx.scene.chart.PieChart;

/** Pie chart for visualized comparison between locations. */
public class PieChartGraph {
	
	/**
	 * The pie chart visualization of data table about confirmed cases
	 * @param countries countries of interest
	 * @param date date of interest
	 * @return array of pie charts for total cases and total cases per 1M
	 * */
	public static Node[] getConfirmedVisualization(String[] countries, Date date) {
		
		 HashMap<String,Long> casesMap = CasesAnalysis.getConfirmedCases(countries, date);
		 HashMap<String,Float> casesPerMMap = CasesAnalysis.getConfirmedCasesPerMil(countries, date);
		 
		 ObservableList<PieChart.Data> casesData=FXCollections.observableArrayList();
		 ObservableList<PieChart.Data> casesPerMData=FXCollections.observableArrayList();
		 
		 for(String country:countries) {
			 long val1=(Long)casesMap.get(country);
			 float val2=(Float)casesPerMMap.get(country);
			 casesData.add(new PieChart.Data(country, val1==-1?0:val1));
			 casesPerMData.add(new PieChart.Data(country, val2==-1?0:val2));
		 }
        final PieChart chartCases = new PieChart(casesData);
        final PieChart chartCasesPerM = new PieChart(casesPerMData);
        SimpleDateFormat sdf=new SimpleDateFormat("MMM dd yyyy");
        chartCases.setTitle("Total Confirmed Cases as of "+sdf.format(date));
        chartCasesPerM.setTitle("Total Confirmed Cases (per 1M) as of "+sdf.format(date));
		
		return new Node[] {chartCases, chartCasesPerM};
	}
}
