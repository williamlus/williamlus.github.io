package comp3111.covid;

/**
 * Class to process Vaccination Table records for TableView.
 * **/
public class TableProcessing3 {
	private String country;
	private Long vaccined;
	private String vaccinedRate;
	
	public TableProcessing3(String country, Long vaccined, String vaccinedRate) {
		this.country = country;
		this.vaccined = vaccined;
		this.vaccinedRate = vaccinedRate;
	}
	
	public String getCountry() {
		return country;
	}
	public Long getVaccined() {
		return vaccined;
	}
	public String getVaccinedRate() {
		return vaccinedRate;
	}
	public void setCountry(String s) {
		country = s;
	}
	public void setVaccined(Long v) {
		vaccined =v;
	}
	public void setVaccinedRate(String f) {
		vaccinedRate =f;
	}
}
