package comp3111.covid;

import java.util.HashMap;
import javafx.scene.chart.CategoryAxis;
import javafx.scene.chart.NumberAxis;
import javafx.scene.chart.StackedBarChart;
import javafx.scene.chart.XYChart;

/** Bar chart for visualized comparison between locations. */
public class BarChartGraph {
	/**
	 * Returns stacked bar chart representation of confirmed death
	 * @param countries the list of countries
	 * @param date the current date
	 **/
	public static StackedBarChart getDeathBarChart(String[] countries, String date) {
		// Set x-axis
		var xAxis = new CategoryAxis();
		// Set the y-axis
        var yAxis = new NumberAxis();
        StackedBarChart <String,Number> bc = new StackedBarChart<String,Number>(xAxis,yAxis);
        bc.setPrefHeight(300);
        bc.setPrefWidth(900);
        bc.autosize();
        bc.setTitle("Total Death Comparison");
        xAxis.setLabel("Country");       
        yAxis.setLabel("Number of Confirmed Deaths");
        
      //Removing the tick labels
        xAxis.setAutoRanging(true);
        yAxis.setAutoRanging(true);
        
     	HashMap<String,Long> hp = CasesAnalysis.getConfirmedDeaths(countries, date);
     	for (String country:countries) {
     		XYChart.Series<String, Number> series = new XYChart.Series<String, Number>();
     		series.setName(country);
     		series.getData().add(new XYChart.Data<String, Number>(country, (Long)hp.get(country))); 
     		bc.getData().add(series);
     	}
        
        return bc;
	}
	/**
	 * Returns stacked bar chart representation of confirmed death per million
	 * @param countries the list of countries
	 * @param date the current date
	 **/
	public static StackedBarChart getDeathBarChartPerMil(String[] countries, String date) {
		// Set x-axis
		var xAxis = new CategoryAxis();
		// Set the y-axis
        var yAxis = new NumberAxis();
        StackedBarChart <String,Number> bc = new StackedBarChart<String,Number>(xAxis,yAxis);
        bc.setPrefHeight(300);

     		bc.getData().add(series);
     	}
        
        return bc;
	}
}
