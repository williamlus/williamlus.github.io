package comp3111.covid;

/**
 * Class to process Confirmed Cases Table records for TableView.
 * **/
public class TableProcessing2 {
	private String country;
	private String confirmed;
	private String confirmedPerMil;
	
	public TableProcessing2(String country, String confirmed, String confirmedPerMil) {
		this.country = country;
		this.confirmed = confirmed;
		this.confirmedPerMil = confirmedPerMil;
	}
	
	public String getCountry() {
		return country;
	}
	public String getConfirmed() {
		return confirmed;
	}
	public String getConfirmedPerMil() {
		return confirmedPerMil;
	}
	public void setCountry(String s) {
		country = s;
	}
	public void setConfirmed(String c) {
		confirmed =c;
	}
	public void setConfirmedPerMil(String f) {
		confirmedPerMil =f;
	}
}
