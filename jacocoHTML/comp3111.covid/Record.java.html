package comp3111.covid;

import java.util.HashMap;
import java.util.Map;

/** Class to store each row of the dataset by its header.
 *  @author LU, Weiqi
 *  */
public class Record{
	private static String[] attributeNames=null; /* All attribute names of the data set */
	private static Map<String,Integer> indexOfName=null; /* The index of an attribute in a record */
	private String[] attributes=null; /* The attribute values of the record */
	
	public Record(String[] record) {
		assert(attributeNames!=null && indexOfName!=null);
		this.attributes=record;
	}
	
	/** Set attribute names of the record according to the header of the data set */
	public static void setAttributeNames(String[] names){
		attributeNames=names;
		Map<String,Integer> map=new HashMap<String,Integer>();
		for(int i=0;i<names.length;i++) {
			map.put(names[i],i);
		}
		indexOfName=map;
		for(String s:indexOfName.keySet()) {
			System.out.print(s+":"+indexOfName.get(s)+", ");
		} System.out.println();
	}
	
	/** Get the attribute of a record by name */
	public String get(String attributeName) {
		assert(attributes!=null);
		int index=getAttributeIndex(attributeName);
		assert(index!=-1);
		return attributes[index];
	}
	
	/** Get the attribute of a record by index */
	public String get(int index) {
		assert(attributes!=null && index>=0 && index<attributes.length);
		return attributes[index];
	}
	
	/** Find the index of an attribute in a record, return -1 if not exists */
	private static int getAttributeIndex(String attributeName) {
		assert(attributeNames!=null && indexOfName!=null);
		return indexOfName.getOrDefault(attributeName, -1);
	}
	
	@Override
	public String toString() {
		String out="";
		for(int i=0;i<attributeNames.length;i++) {
			out+=(attributeNames[i]+":"+attributes[i]+"; ");
		}
		return out;
	}
}
